        function updateHamsters() {

            const now = Date.now();

            const deltaTime = (now - gameState.lastUpdate) / 1000;


            gameState.hamsters.forEach(hamster => {

                hamster.progress = Math.min(100, hamster.progress + (deltaTime * 2));

            });


            gameState.lastUpdate = now;

            renderHamsters();

            updateDisplay();

        }


        function collectFromHamster(hamsterId) {

            const hamster = gameState.hamsters.find(h => h.id === hamsterId);

            if (hamster && hamster.progress > 0) {

                const coinsEarned = Math.floor(hamster.progress * hamster.miningRate / 10);

                gameState.coins += coinsEarned;

                gameState.totalEarned += coinsEarned;

                hamster.progress = 0;

                

                showFloatingCoins(coinsEarned);

                showNotification(`${hamster.name} earned ${coinsEarned}💰!`);

                

                // Haptic feedback for Telegram

                if (tg) tg.HapticFeedback.impactOccurred('light');

                

                renderHamsters();

                updateDisplay();

                saveGame();

            }

        }


        function collectAll() {

            let totalEarned = 0;

            gameState.hamsters.forEach(hamster => {

                if (hamster.progress > 0) {

                    const coinsEarned = Math.floor(hamster.progress * hamster.miningRate / 10);

                    totalEarned += coinsEarned;

                    hamster.progress = 0;

                }

            });


            if (totalEarned > 0) {

                gameState.coins += totalEarned;

                gameState.totalEarned += totalEarned;

                showFloatingCoins(totalEarned);

                showNotification(`Collected ${totalEarned}💰 from all hamsters!`);

                

                // Haptic feedback

                if (tg) tg.HapticFeedback.impactOccurred('medium');

                

                renderHamsters();

                updateDisplay();

                saveGame();

            }

        }


        function boostAll() {

            if (gameState.coins >= 100) {

                gameState.coins -= 100;

                gameState.speedMultiplier = Math.min(5, gameState.speedMultiplier + 0.5);

                showNotification(`⚡ Speed boosted! ${gameState.speedMultiplier}x faster!`);

                

                if (tg) tg.HapticFeedback.impactOccurred('heavy');

                

                updateDisplay();

                saveGame();

                

                setTimeout(() => {

                    gameState.speedMultiplier = Math.max(1, gameState.speedMultiplier - 0.5);

                    showNotification('⚡ Speed boost expired');

                    updateDisplay();

                    saveGame();

                }, 30000);

            } else {

                showNotification('❌ Need 100💰 for boost!');

                if (tg) tg.HapticFeedback.notificationOccurred('error');

            }

        }


        function levelUpHamster(hamsterId) {

            const hamster = gameState.hamsters.find(h => h.id === hamsterId);

            const cost = hamster.level * 100;

            

            if (gameState.coins >= cost) {

                gameState.coins -= cost;

                hamster.level++;

                hamster.miningRate += 2;

                showNotification(`🔥 ${hamster.name} → Level ${hamster.level}!`);

                

                if (tg) tg.HapticFeedback.impactOccurred('medium');

                

                renderHamsters();

                updateDisplay();

                saveGame();

            } else {

                showNotification(`❌ Need ${cost}💰 to level up ${hamster.name}`);

                if (tg) tg.HapticFeedback.notificationOccurred('error');

            }

        }


        function buyHamster(type) {

            const prices = { bitcoin: 500, ethereum: 1000, dogecoin: 300 };

            const emojis = { bitcoin: '🟡', ethereum: '💎', dogecoin: '🐕' };

            const rates = { bitcoin: 5, ethereum: 8, dogecoin: 3 };

            

            const cost = prices[type];

            

            if (gameState.coins >= cost) {

                gameState.coins -= cost;

                const newHamster = {

                    id: Date.now(),

                    name: `${type.charAt(0).toUpperCase() + type.slice(1)} Miner`,

                    type: type.charAt(0).toUpperCase() + type.slice(1),

                    emoji: emojis[type],

                    level: 1,

                    miningRate: rates[type],

                    progress: 0,

                    lastCollected: Date.now()

                };

                

                gameState.hamsters.push(newHamster);

                showNotification(`🎉 Bought ${type} hamster!`);

                

                if (tg) tg.HapticFeedback.impactOccurred('heavy');

                

                renderHamsters();

                updateDisplay();

                saveGame();

            } else {

                showNotification(`❌ Need ${cost}💰!`);

                if (tg) tg.HapticFeedback.notificationOccurred('error');

            }

        }


        function buyUpgrade(type) {

            if (type === 'speed' && gameState.coins >= 200) {

            }

        }
